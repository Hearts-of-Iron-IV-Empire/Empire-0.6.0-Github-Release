RUS_dont_call_in_zap_to_gtw = { # Prevents Russia from calling in Zaparozhia to the Great Turkish War for historical reasons
	allowed = {
		original_tag = SOV
	}
	enable = {
	    country_exists = ZAP
		has_country_flag = dont_call_in_zap_to_gtw
	}
	abort = {
		NOT = {

		}
	}
	ai_strategy = {
		type = diplo_action_desire
		id = ZAP
		value = -20000
		target = call_allies
	}
}
